147 uint8_t OMMoCoBus::getPacket() {
148 
149  uint8_t command_val = 0;
150  uint8_t com_byte_count = 0;
151  //uint8_t dat = 0;
152  uint8_t ret = 0;
153 
154  m_isBCast = false;
155 
156  bool not_us = false;
157 
158  // we check serial.available() here even
159  // though this->_getNextByte() does, because
160  // a timeout is applied waiting for next command
161  // in this->_getNextByte() - since this is called
162  // every cycle of the main loop, we don't want
163  // to introduce a delay unless we know of a
164  // byte waiting to be read
165 
166  int avail = m_serObj->available();
167 
168  // serial does not have data in the buffer
169  if( avail <= 0 )
170  return(0);
171 
172  // is this packet intended for us?
173  uint8_t stat = this->_targetUs();
174 
175  if(stat == OM_SER_ERR || stat == OM_SER_TIMEOUT) {
176  // got an error? timeout error, really.
177  // this->_flushSerial();
178  return(0);
179  }
180  else if( stat == OM_SER_IS_BCAST ) {
181  // command was a broadcast command
182  m_isBCast = true;
183  }
184  else if( stat != OM_SER_OK ) {
185  // command was for someone else
186  // (but we need to read the rest of the data
187  // off the line to make sure we don't keep part of it
188  // in the buffer)
189  not_us = true;
190  }
191 
192  // get command byte
193  ret = this->_getNextByte(command_val);
194 
195  if ( ret != OM_SER_OK ) {
196  // well, this is just embarassing
197  // this->_flushSerial();
198  return(0);
199  }
200 
201  if( command_val > 0 ) {
202  // get data length (even if not our own command)
203  ret = this->_getNextByte(com_byte_count);
204 
205  if ( ret != OM_SER_OK ) {
206  // well, this is just embarassing
207  // this->_flushSerial();
208  return(0);
209  }
210  }
211 
212  // check for overflow
213  // note: we will still be in error condition, as
214  // there will be values left in the serial buffer
215  // on the next read, from this command - so don't
216  // exceed it, eh? This is just to prevent memory
217  // corruption. It is not safe otherwise
218 
219  if(com_byte_count > OM_SER_BUFLEN)
220  com_byte_count = OM_SER_BUFLEN;
221 
222 
223  // clear out any previous command data
224  memset(m_serBuffer, 0, sizeof(uint8_t) * OM_SER_BUFLEN);
225  m_bufSize = 0;
226 
227 
228  // populate command data buffer
229  for( int i = 1; i <= com_byte_count; i++ ) {
230  // get com_byte_count character values from the serial
231  // buffer
232 
233  uint8_t dat = 0;
234  uint8_t ret = this->_getNextByte(dat);
235 
236  if( ret != OM_SER_OK )
237  return(0);
238 
239  m_serBuffer[i - 1] = dat;
240  m_bufSize++;
241  }
242 
243  if( not_us == true ) {
244  // the command was not intended for us
245  memset(m_serBuffer, 0, sizeof(uint8_t) * OM_SER_BUFLEN);
246  return(0);
247  }
248 
249 
250  return( command_val );
251 }

339 void OMMoCoBus::sendPacketHeader(unsigned int p_addr, uint8_t p_code, uint8_t p_dlen) {
340 
341  if( p_addr == OM_SER_BCAST_ADDR )
342  m_isBCast = true;
343  else
344  m_isBCast = false;
345 
346  // start sequence of five nulls
347  for( uint8_t i = 0; i <= 4; i++ ) {
348  this->write((uint8_t) 0);
349  }

351  // start sequence termination
352  this->write((uint8_t) 255);
353 
354  // target address
355  this->write(p_addr);
356 
357  // command code or response code
358  this->write(p_code);
359

513 uint8_t OMMoCoBus::_targetUs() {
514 
515  uint8_t thsChar = 1;
516  uint8_t ret = this->_getNextByte(thsChar);
517 
518  // timeout? other error?
519  if( ret != OM_SER_OK )
520  return(OM_SER_TIMEOUT);
521 
522  // if character is null...
523  if( thsChar == 0 ) {
524  // until we get enough nulls for a start command
525  // or read something that isn't null...
526 
527  uint8_t nullCount = 0;
528 
529  while( nullCount < 5 && thsChar == 0 ) {
530  nullCount++;
531 
532  // null count met
533  if( nullCount == 5 )
534  break;
535 
536  ret = this->_getNextByte(thsChar);
537 
538  if( ret != OM_SER_OK )
539  return(OM_SER_TIMEOUT);
540 
541  }
542 
543  // did not receive a proper start sequence
544  if( nullCount < 5 )
545  return(OM_SER_ERR);
546 
547  thsChar = 0;
548 
549  ret = this->_getNextByte(thsChar);
550 
551  if( ret != OM_SER_OK )
552  return(OM_SER_TIMEOUT);
553 
554  if( thsChar != 255 )
555  return(OM_SER_ERR);
556 
557  // get target address
558 
559  unsigned int addr = 0;
560  ret = this->_getNextByte(thsChar);
561 
562  if( ret != OM_SER_OK )
563  return(OM_SER_TIMEOUT);
564 
565  addr = (unsigned int) thsChar << 8;
566 
567  ret = this->_getNextByte(thsChar);
568 
569  if( ret != OM_SER_OK )
570  return(OM_SER_TIMEOUT);
571 
572  addr |= (unsigned int) thsChar;
573 
574  // is not destined for us?
575 
576  if( addr == OM_SER_BCAST_ADDR )
577  return(OM_SER_IS_BCAST);
578  else if( addr != m_devAddr )
579  return(OM_SER_NOT_US);
580 
581  } // end if(thsChar == 0
582  else {
583  // uint8_t invalid
584  return(OM_SER_ERR);
585  }
586 
587  return(OM_SER_OK);
588 
589 }

591 uint8_t OMMoCoBus::_getNextByte(uint8_t& p_data) {
592 
593  uint8_t ret = OM_SER_OK;
594  uint8_t* p = (uint8_t*)(void*) &p_data;
595 
596  if( m_serObj->available() > 0 ) {
597  *p = (uint8_t) m_serObj->read();
598  }
599  else {
600  unsigned long timer = millis();
601  // wait until we have data ready
602  while( m_serObj->available() < 1 ) {
603  if( millis() - timer > OM_SER_WAIT ) {
604  // timed out waiting for complete input
605  return( OM_SER_TIMEOUT );
606  }
607  }
608  *p = (uint8_t) m_serObj->read();
609  }
610 
611 
612  return(ret);
613 }
