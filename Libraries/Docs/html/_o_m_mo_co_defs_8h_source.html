57 // serial bits per second on the bus
58 // 57.6k is the maximum bit rate without risking
59 // the high probability of framing errors
60 // this number must be a floating point!
61 
62 #define OM_SER_BPS 57600.0

64 // serial clear time in uS (how long it takes to punch one bit
65 // out through the RS485 transceiver at the current bitrate).
