375 int OMMoCoMaster::_getResponse() {
376 
377  // if we sent a broadcast packet, do not look for a response
378  if( isBroadcast() == true )
379  return(1);
380 
381  unsigned long cur_tm = millis();
382 
383  // timeout if we don't get a response packet in
384  // time
385 
386  uint8_t code = getPacket();
387  while (code == 0) {
388  if (millis() - cur_tm > OM_SER_MASTER_TIMEOUT)
