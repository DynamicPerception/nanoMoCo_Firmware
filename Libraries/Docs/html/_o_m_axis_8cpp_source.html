325 bool OMAxis::moveMotor(bool dir, uint32_t steps,
326  uint32_t totalTravelTime, uint32_t accelTime, uint32_t decelTime) {
327  OMByteBuffer sendBuf;
328  uint8_t _buf[22];
329  sendBuf.assign(_buf, sizeof(_buf));
330  sendBuf.append((uint8_t) CMD_PC_MOVE_MOTOR_CPLX);
331  sendBuf.append(dir);
332  sendBuf.append(htonl(steps));
333  sendBuf.append((uint8_t)OM_PDU_ESC);
334  sendBuf.append(htonl(totalTravelTime));
335  sendBuf.append((uint8_t)OM_PDU_ESC);
336  sendBuf.append(htonl(accelTime));
337  sendBuf.append((uint8_t)OM_PDU_ESC);
338  sendBuf.append(htonl(decelTime));
339  int res = command(m_slaveAddr, OM_PCODE_PC, sendBuf.getBuf(),
340  sendBuf.size());
341  if (res == 0)
342  return true;
343  else
344  return false;
345 }
