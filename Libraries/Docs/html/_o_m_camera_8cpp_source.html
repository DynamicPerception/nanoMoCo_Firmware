205 void OMCamera::stop() {
206 
207  // stop timer (always)
208  // we do this before bringing the pin low in case
209  // a VERY short action time was set to prevent the
210  // timer from continuing to trigger and getting stuck in
211  // a loop.
212 
213  MsTimer2::stop();
214 
215  uint8_t code = 0;
216 
217  if( m_curAct == OM_CAM_INEXP ) {
218  // expose complete
219  digitalWrite(m_shutter, LOW);
220 
221  // we do this every time, because
222  // it's possible that the flag
223  // that controls whether or not to
224  // trip focus w. shutter may have
225  // been reset during our exposure,
226  // and failing to do so would keep
227  // the focus pin high for a long
228  // time.
229 
230  digitalWrite(m_focus, LOW);
231  code = OM_CAM_EFIN;
232  }
233  else if( m_curAct == OM_CAM_INFOC ) {
234  // focus complete
235  digitalWrite(m_focus, LOW);
236  code = OM_CAM_FFIN;
237  }
238  else {
239  code = OM_CAM_WFIN;
240  }
241 
242 
243  // reset current action
244  m_curAct = OM_CAM_INCLR;
245 
246  // update camera currently engaged
247  m_isBzy = false;
248 
249 
250  if( code != 0 ) {
251  // if we got here because of stop() called from a timer, then
252  // let's send a proper signal (if callback set)
253  if( f_camSignal != 0 )
254  f_camSignal(code);
255  }
256 
257 }

281 void OMCamera::expose(unsigned long p_Time) {
282 
283  // do not expose if exposure time is zero
284  if( p_Time == 0 ) {
285  if( f_camSignal != 0 )
286  f_camSignal(OM_CAM_EFIN);
287 
288  return;
289  }
290  // indicate (for stop()), that we are in an exposure
291  // to properly set output states when timer2 completes
292 
293  m_curAct = OM_CAM_INEXP;
294 
295 
296  // determine if focus pin should be brought high
297  // w. the shutter pin (for some nikons, etc.)
298 
299  if( m_focusShut == true )
300  digitalWrite(m_focus, HIGH);
301 
302  digitalWrite(m_shutter, HIGH);
303 
304  // start timer to stop camera exposure
305  MsTimer2::set(p_Time, OMCamera::stop);
306  MsTimer2::start();
307 
308 
309  // update camera currently engaged
310  m_isBzy = true;
311 
312 
313  // if there is a pointer to a function
314  // to be called when the camera is done,
315  // do it now
316 
317  if( f_camSignal != 0 )
318  f_camSignal(OM_CAMEXP);
319 
320  return;
321 }

345 void OMCamera::focus(unsigned int p_Time) {
346 
347  // do not focus if focus time is 0
348  if( p_Time == 0 ) {
349  if( f_camSignal != 0 )
350  f_camSignal(OM_CAM_FFIN);
351 
352  return;
353  }
354 
355  digitalWrite(m_focus, HIGH);
356 
357  // start timer to stop focus engage
358 
359  m_curAct = OM_CAM_INFOC;
360 
361  MsTimer2::set(p_Time, OMCamera::stop);
362  MsTimer2::start();
363  // update camera currently engaged
364  m_isBzy = true;
365  // report focus in progress
366  if( f_camSignal != 0 )
367  f_camSignal(OM_CAMFOC);
368 
369 }
