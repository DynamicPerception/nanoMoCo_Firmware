1061 void OMMotor::move(bool p_Dir, unsigned long p_Steps) {
1062 
1063 
1064  // if motor is disabled, do nothing
1065  if( ! enable() || ( maxSteps() > 0 && stepsMoved() >= maxSteps() ) ) {
1066  _fireCallback(OM_MOT_DONE);
1067  return;
1068  }
1069 
1070  // note: the check on p_Steps is required
1071  // to allow manual moves that are not continuous
1072  // when motor is set in continuous mode
1073 
1074  if( p_Steps == 0 && continuous() ) {
1075  // continuous motion mode
1076  if( ! running() ) {
1077  m_asyncSteps = 0;
1078  _stepsAsync(p_Dir, (unsigned long) 0);
1079  }
1080 
1081  _fireCallback(OM_MOT_BEGIN);
1082  return;
1083  }
1084 
1085  // continuous allows higher stepping rate
1086  // so clamp back down if not doing a continuous
1087  // move
1088 
1089  if( maxStepRate() > 5000 )
1090  maxStepRate(5000);
1091 
1092  // check for backlash compensation
1093  if( m_backCheck == true ) {
1094  p_Steps += backlash();
1095  m_backCheck = false;
1096  }
1097 
1098  // not in continuous motion mode
1099  // nothing to do
1100  if( p_Steps == 0 ) {
1101  _fireCallback(OM_MOT_DONE);
1102  return;
1103  }
1104 
1105  if( ! m_calcMove ) {
1106 
1107  // this move was not specified with a time to accomplish, so accomplish
1108  // in minimal amount of time possible
1109 
1110  // determine how far we'd move in our acceleration and deceleration
1111  // times for moves that must operate at maximum speed. We accelerate
1112  // and decel over 200 steps each, or 1/4 of the total steps, whichever is less
1113 
1114  // set travel const (.travel) here before attempting to use it
1115  _setTravelConst(&m_splineOne);
1116 
1117  unsigned int mSpeed = ( maxStepRate() > maxSpeed() )? maxSpeed() : maxStepRate();
1118 
1119  float rampSteps = (200 > (p_Steps / 4)) ? (p_Steps / 4) : 200;
1120 
1121  rampSteps *= 2;
1122 
1123  float crTm = ((p_Steps - rampSteps) / mSpeed) * 1000;
1124  float adTm = ((rampSteps / mSpeed) * 1000) * m_splineOne.travel;
1125 
1126  float mvMS = (crTm + adTm) + 1.0;
1127 
1128  // take a minimum of 50ms to make the move - to prevent over-speeding
1129  // and getting goofy.
1130  if( mvMS < 50.0 ) {
1131  mvMS = 50.0;
1132  }
1133 
1134 
1135  // one spline point per ms, like normal.
1136  m_totalSplines = (unsigned long) mvMS;
1137 
1138  // prep spline variables
1139  _initSpline(false, p_Steps, mvMS, adTm/2, adTm/2);
1140 
1141  // we need to initialize the first spline point
1142  m_curSpline = 1;
1143  float tmPos = (float) m_curSpline / (float) m_totalSplines;
1144  // calculate the first running speed
1145  f_easeFunc(false, tmPos);
1146  }
1147 
1148  // limit step cycle using this control
1149  // value, without mucking with max steps
1150 
1151  m_asyncSteps = p_Steps;
1152 
1153  // re-set called from five-argument flag
1154 
1155  m_calcMove = false;
1156 
1157  _stepsAsync(p_Dir, p_Steps);
1158  _fireCallback(OM_MOT_MOVING);
1159 
1160 }

1307 void OMMotor::_runISR() {
1308 
1309 
1310  // digital pin 9
1311  static float cycleErrAccumulated = 0.0;
1312  static unsigned long cyclesLow = 0;
1313  static unsigned long stepsTaken = 0;
1314  static unsigned long totalCyclesTaken = 0;
1315 
1316 
1317  if( m_asyncSteps > 0 && totalCyclesTaken >= m_cyclesPerSpline ) {
1318  // we are ready for the next point in the spline,
1319  // run speed calculations
1320 
1321  if( m_asyncSteps > 0 && m_curSpline == m_totalSplines ) {
1322  // hey, look at that - we're at the end of our spline (and
1323  // we haven't finished our last step either, otherwise we
1324  // wouldn't get here...)
1325 
1326  if( stepsTaken < m_asyncSteps ) {
1327  // we really should be taking any steps we're missing. (if asked to
1328  // move a specific distance.) positioning errors can add up over
1329  // multiple moves, and we shouldn't leave expected steps on the
1330  // cutting-room floor even if we have exceeded the destination time
1331 
1332  // TODO: multiple final steps will happen at the fastest step rate,
1333  // whether or not the motor can handle this! We really shouldn't be
1334  // more than one step shy of our target, however, unless one specifies
1335  // a move that requires a step rate higher than the maximum step rate,
1336  // which is just painful.
1337 
1338  // digital pin 9 on atmega328p
1339  PORTB |= B00000010;
1340 
1341  totalCyclesTaken--;
1342  stepsTaken++;
1343  _updateMotorHome( (uint8_t) 1);
1344  // bring step pin low - this allows us to
1345  // hit one step per ISR run rate.
1346  PORTB &= B11111101;
1347  return;
1348  }
1349 
1350  // we've reached the end of our anticipated time. Go ahead and finish
1351  // everything up, and stop the timer. (we should only get here when a
1352  // last step had to be taken above.)
1353 
1354  stepsTaken = 0;
1355  m_curSpline = 0;
1356  totalCyclesTaken = 0;
1357  cycleErrAccumulated = 0.0;
1358  cyclesLow = 0;
1359 
1360  stop();
1361  return;
1362  } // end if( m_curSpline...
1363 
1364  // move to the next point in the current spline.
1365  m_curSpline++;
1366  float tmPos = (float) m_curSpline / (float) m_totalSplines;
1367 
1368  // get new off cycle timing for the next point in the spline
1369  f_easeFunc(false, tmPos);
1370  totalCyclesTaken = 0;
1371 
1372  // we don't stop here, as we still need to check the off-time
1373  // between steps, based on our current spline point
1374 
1375  } // end if( totalCyclesTaken...
1376 
1377 
1378  totalCyclesTaken++;
1379 
1380  if( cycleErrAccumulated >= 1.0 ) {
1381 
1382  // check the error rate (fractions of a cycle that
1383  // were accumulated - if at least one full cycle, add
1384  // an additional delay cycle to get overall movement
1385  // timing as close to exact as is possible
1386 
1387  cycleErrAccumulated -= 1.0;
1388  // run an extra cycle low
1389  return;
1390  }
1391 
1392  // increase lowcycle counter after applying error correction, as we
1393  // don't want to take the next step too fast
1394 
1395  cyclesLow++;
1396 
1397  if( cyclesLow >= m_curOffCycles ) {
1398  // we've had enough low cycles, ok to trigger next step
1399 
1400  // set pin to trigger step
1401  // digital pin 9 on atmega328p
1402  PORTB |= B00000010;
1403 
1404  cyclesLow = 0;
1405  stepsTaken++;
1406 
1407  _updateMotorHome(1);
1408 
1409  // accumulate cycle off time errors
1410  // once per complete cycle (fractions of
1411  // off cycles are included in the calculated
1412  // speed)
1413 
1414  cycleErrAccumulated += m_curCycleErr;
1415 
1416  // if we hit the step count requested for this move,
1417  // or if we have hit the maximum stepping point,
1418  // stop now - don't overshoot
1419  if( (m_totalSteps > 0 && m_stepsMoved >= m_totalSteps) || (m_asyncSteps > 0 && stepsTaken >= m_asyncSteps) ) {
1420  stepsTaken = 0;
1421  cycleErrAccumulated = 0.0;
1422  cyclesLow = 0;
1423  stop();
1424  }
1425 
1426  // bring step pin low again, the driver will register
1427  // a pulse of only a few uS, so we can perform
1428  // a complete step in a single timing cycle,
1429  // allowing us to hit a speed of one step per
1430  // timing cycle
1431 
1432  PORTB &= B11111101;
1433 
1434  } // end if( cyclesLow...
1435 
1436 
1437 }

1494 void OMMotor::home() {
1495  // we don't want to be in continuous mode when
1496  // performing a send to home, so that the logic
1497  // will make one (busy) move.
1498 
1499  bool wasCont = continuous();
1500  bool thsDir = false;
1501 
1502  continuous(false);
1503 
1504  // negative value means move in
1505  // positive direction
1506 
1507  if( m_homePos < 0 ) {
1508  m_homePos *= -1;
1509  thsDir = true;
1510  }
1511 
1512  // there is no need to re-set m_homePos, as travelling in the
1513  // specified direction will correct the recorded steps
1514 
1515  move(thsDir, m_homePos);
1516 
1517  continuous(wasCont);
1518 
1519 }

1521 /*
1522 
1523  This linear easing algorithm calculates the current delay cycles between steps,
