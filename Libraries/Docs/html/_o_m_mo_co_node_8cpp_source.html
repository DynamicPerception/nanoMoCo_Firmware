276 uint8_t OMMoCoNode::check() {
277 
278 
279 
280  uint8_t command = getPacket();
281 
282  // no packet available
283  if( command == 0 )
284  return(0);
285 
286  // command OM_SER_BASECOM is reserved for core protocol commands.
287  // We handle these automatically for the node.
288 
289  if( ! isBroadcast() && command == OM_SER_BASECOM ) {
290  _coreProtocol(this->buffer());
291  return(0);
292  }
293  else {
294  // we have a command code which is higher than 1,
295  // send to the supplied handler.
296  if( f_bcastHandler != 0 && isBroadcast() ) {
297  // a callback is assigned, and this is a
298  // broadcast command
299  f_bcastHandler(command, this->buffer());
300  }
301  else if( f_cmdHandler != 0 && ! isBroadcast() ) {
302  // a callback is assigned, call callback
303  // with command code and payload
304  f_cmdHandler(command, this->buffer());
305  }
306  }
307 
308 
309  return(command);
310 }

332 void OMMoCoNode::_coreProtocol( uint8_t* p_buf ) {
333  byte subCom = p_buf[0];
334 
335  switch( subCom ) {
336  case OM_SER_COREPROTO:
337  // core protocol version supported
338  response(true, (unsigned int) OM_SER_VER);
339  break;
340  case OM_SER_COREID:
341  // device identifier
342  response(true, id(), 8);
343  break;
344  case OM_SER_COREVER:
345  // device version
346  response(true, m_ver);
347  break;
348  case OM_SER_COREADDR:
349  // change node addr
350  address(p_buf[1]);
351  response(true);
352  break;
353  default:
354  // error
355  response(false);
356  }

358 }
