44 void OMComHandler::_masterFollow() {
45  // slave, attach an interrupt to the
46  // common line, engage pull-up resistor
47  pinMode(OM_COM1, INPUT);
48  digitalWrite(OM_COM1, HIGH);
49  attachInterrupt(0, _slaveTripped, FALLING);
50 }

52 void OMComHandler::_masterLead() {
53  // master, bring the common line high (bring low to signal slaves)
54  detachInterrupt(0);
55  pinMode(OM_COM1, OUTPUT);
56  digitalWrite(OM_COM1, HIGH);
57 }

116 void OMComHandler::masterSignal() {
117  if( this->m_isMaster ) {
118  // we're the master, send a 5mS low pulse on
119  // common line

121  delay(5);
